Modal - A Bootstrap 4.0+ jQuery plugin for the modal component

Simple Example
$.modal({
    type: "success", // "info", "success", "warning", "error"
    title: "Awesome!",
    content: "The action has been completed."
});
Title Only Example
$.modal({
    type: "success", // "info", "success", "warning", "error"
    title: "Awesome!"
});
Content Only Example
$.modal({
	type: "success",
  	content: "The action has been completed."
});
Buttons Example
$.modal({
	type: "success", // "info", "success", "warning", "error",
	title: "Awesome!",
  	content: "The action has been completed.",
    buttons: [
    	{
    		type: "button", // "submit", "button",
    		label: "Click Me!",
			callback: function () {
      			alert("Hey! You clicked me.");
    		}
    	},
		{
			type: "dismiss", // dismiss will hide the modal 
    		label: "Close"
  		}
	] 
});
Sizes Example
$.modal({ 		
    type: "success",
  	title: "Awesome!",
  	content: "The action has been completed.",
  	size: "lg" // "sm", "" (default), "lg", "xl"
});
